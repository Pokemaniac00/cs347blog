If I were to demonstrate what I learned from this page that helped with my lack of understanding from class was how states are used for ReactJS code. Take a look at the code sample below. We can observe that this class extends the react component class, and has multiple methods, including an instructor that initializes a state with a quote and an author. The state is then set in handleClick(), also managing states. Quotes would be returned from an array of quotes and authors, selected at random using a random number generator. 

                
class App extends React.Component {
    constructor() {
        super();
        this.state = {
        quote: quotes[0].quote,
        author: quotes[0].author,
        };
    }
    randomQuote() {
        const randomNumber = Math.floor(Math.random() * quotes.length);
        return quotes[randomNumber];
        
    }
    shuffleQuotes(array){
        return array.sort(()=>Math.random()-0.5)
    }
    
    handleClick = () => {
        const generateRandomQuote = this.randomQuote();
        this.setState({
        quote: generateRandomQuote.quote,
        author: generateRandomQuote.author
        });
        this.shuffleQuotes(quotes)
    };
    
    randomColor() {
        const color = `rgb(
        ${Math.floor(Math.random() * 155)},
        ${Math.floor(Math.random() * 155)},
        ${Math.floor(Math.random() * 155)})`;
        return color;
    }
    
    render() {
        return (
            <div>
            <QuoteAndAuthor
            displayColor={this.randomColor}
            handleClick={this.handleClick}
            {...this.state}
            />
            </div>
        );
    }
